Core-Table ​

结合el-table和el-pagantion 组合封装的table组件 可以搭配Hooks-useQueryList使用

html
<core-table v-loading="pageLoading" :coloum="coloumList" :data="dataList">
	// 自定义coloum
	<template #uavTypesZh="{ scope }"><el-tag>{{ scope.row.uavTypesZh }}</el-tag></template>
	<template #modify="{ scope }">
		<el-button size="small">修改</el-button>
		<el-button size="small">删除</el-button>
	</template>
</core-table>
javascript
import { ref } from 'vue'
import { FormComponentType, createColoum, createForm, createMenu } from '@/components/core-element/utils'

const dataList = [
	{
		deptName: '场站1',
		airspaceTypeZh: 'asdas',
		uavTypesZh: '飞机类型222',
		fileStatusZh: '通过'
	}
]

const coloumList = ref(
	createColoum([
		{
			label: '场站',
			prop: 'deptName'
		},
		{
			label: '空域类型',
			prop: 'airspaceTypeZh'
		},
		{
			label: '飞机类型',
			prop: 'uavTypesZh'
		},
		{
			label: '批文状态',
			prop: 'fileStatusZh'
		},
		{
			label: '操作',
			prop: 'modify'
		}
	])
)
