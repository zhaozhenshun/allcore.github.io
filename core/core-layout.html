Core-Layout ​

布局组件 分上下 左右 可连续嵌套使用

垂直布局 ​

上部不固定、下部内容自由拉伸 总高固定100%

html
<template>
  <CoreLayout layout="vertical">
    <template #top>
      <div style="background-color: rgb(54, 93, 207)">
        <core-form :labelWidth="120" :col="6" :list="formConfig">
          <template #search="{ formInstance, formData }">
            <el-col>
              <el-row class="mb-20">
                <el-col :span="12">
                  <el-button :type="'primary'" @click="addFan">新增</el-button>
                  <el-button :type="'primary'" @click="handleBatch">批量上传</el-button>
                  <el-button @click="handleDown">下载模板</el-button>
                </el-col>
                <el-col :span="12">
                  <el-row justify="end">
                    <el-button @click="search(formInstance, formData)">查询</el-button>
                    <el-button @click="reset(formInstance, formData)">重置</el-button>
                  </el-row>
                </el-col>
              </el-row>
            </el-col>
          </template>
        </core-form>
      </div>
    </template>

    <template #bottom>
      <core-table v-loading="pageLoading" :coloum="coloumList" :data="result" :pagination="pagination">
        <template #fanNum="{ scope }">
          <el-tag>{{ scope.row.fanNum }}</el-tag>
        </template>
        <template #modify="{ scope }">
          <!-- <el-button size="small" type="modify" @click="handleModify(scope.row)">修改</el-button>
          <el-button size="small" @click="deleteFile(scope.row)">删除</el-button> -->
          <core-menu :menu="getMenu(scope.row)" @menu-click="evt => handleMenu(evt, scope.row)" />
        </template>
      </core-table>
    </template>

  </CoreLayout>
</template>
